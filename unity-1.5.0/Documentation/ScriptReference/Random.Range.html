Random.Range

float Range (float min, float max)

Returns a random number between and including min and max

// Instantiates prefab somewhere between -10 and 10 on the x-z plane var prefab : GameObject;
function Start () {
var position = Vector3(Random.Range(-10, 10), 0, Random.Range(-10, 10));
Instantiate(prefab, position, Quaternion.identity);
}

Random.Range

int Range (int min, int max)

Returns a random number starting between min and max. That is the returned number will never be max

// Loads a random level from the level list Application.LoadLevel(0, Application.levelCount);
