static var toolbar : GUIStyle

Description

Toolbar background from top of windows.

The toolbar is used as a background for other toolbar elements

GUILayout.BeginHorizontal (EditorStyles.toolbar);
GUILayout.Button ("Hello", EditorStyles.toolbarButton);
GUILayout.Button ("Pop up", EditorStyles.toolbarPopup);
EditorGUILayout.Space (); // Get in standard editor element spacing
GUILayout.Button ("Drop me down", EditorStyles.toolbarDropDown);
GUILayout.EndHorizontal ();
