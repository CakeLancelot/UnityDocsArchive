Texture2D.frame

int frame

Currently active frame in the texture. Change this value to update how the texture looks. The new frame will be uploaded to the graphic card, so this is a potentially expensive operation.

var texture : Texture2D;
var framesPerSecond = 12.0;

function Update () {
// Calculate the frame to use by repeating time when it exceeds playLength frameCount = texture.frameCount;
playLength = frameCount / framesPerSecond;
time = Mathf.Repeat (Time.time, playLength) / playLength;
var frame : int = time * frameCount;
frame = Mathf.Clamp (frame, 0, texture.frameCount -1);
texture.frame = frame;
}

function Reset () {
if (renderer) {
texture = renderer.sharedMaterial.mainTexture;
}
}
