var active : bool

Description

Specifies if the keyboard is visible or is sliding into the position on the screen.

Use this property to bring previously hidden keyboard back on the screen.

/// Hides the keyboard if the device is facing down and resumes input if the device is facing up
private var keyboard : iPhoneKeyboard;
function Update() {
if (keyboard && iPhoneInput.orientation == iPhoneOrientation.FaceDown)
keyboard.active = false;
if (keyboard && iPhoneInput.orientation == iPhoneOrientation.FaceUp)
keyboard.active = true;
}
function OnGUI() {
if (GUI.Button(Rect(0, 10, 200, 32), "Open keyboard"))
keyboard = iPhoneKeyboard.Open("text");
}
