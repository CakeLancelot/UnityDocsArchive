Class Variables

Random.value

float value

A random number between 0 and 1

// Prints something between 0 and 1
print(Random.value);

Random.insideUnitSphere

Vector3 insideUnitSphere

A random point inside a sphere with radius 1

// Sets the position to be somewhere inside a sphere
// with radius 5 and the center at zero.
transform.position = Random.insideUnitSphere * 5;

Random.onUnitSphere

Vector3 onUnitSphere

A random point on the surface of a sphere with radius 1

// Sets the rigidbody velocity to 10 and in a random direction.
rigidbody.velocity = Random.onUnitSphere * 10;

Random.rotation

Quaternion rotation

A random rotation

// Instantiates a prefab with a random rotation at the world-origin
var prefab : GameObject;
Instantiate(prefab, Vector3.zero, Random.rotation);

Class Methods

Random.Range

float Range (float min, float max)

Returns a random number between and including min and max

// Instantiates prefab somewhere between -10 and 10 on the x-z plane
var prefab : GameObject;
function Start () {
    var position = Vector3(Random.Range(-10, 10), 0, Random.Range(-10, 10));
    Instantiate(prefab, position, Quaternion.identity);
}

Random.Range

int Range (int min, int max)

Returns a random number starting between min and max. That is the returned number will never be max

// Loads a random level from the level list
Application.LoadLevel(0, Application.levelCount);
