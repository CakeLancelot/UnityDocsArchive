EditorUtility.FindAsset

Object FindAsset (string path, Type type)

Finds an asset at path name. The pathname must include the file extension. It should not be prefixed with "Assets". This returns only assets that are visible in the project view.

// Javascript example: import UnityEditor;

@MenuItem ("GameObject/Assign Default Skybox")
static function AssignSkybox () {
var sky = Camera.main.GetComponent(Skybox);
sky.material = EditorUtility.FindAsset(
"Standard Assets/Skyboxes/ Tropical.mat", Material);
}

@MenuItem ("GameObject/Create First Person Controller")
static function BuildFPS () {
var prefab = EditorUtility.FindAsset(
"Standard Assets/ Prefabs/First Person Controller.prefab",
Transform);

var fps = Instantiate(prefab);
Selection.activeObject = fps;
}

// C# Example: [MenuItem ("GameObject/Assign Default Skybox")]
static void AssignSkybox () {
Skybox sky = (Skybox)Camera.main.GetComponent(typeof(Skybox));
sky.material = (Material)EditorUtility.FindAsset(
"Standard Assets/Skyboxes/ Tropical.mat", typeof(Material));
}

[MenuItem ("GameObject/Create First Person Controller")]
static void BuildFPS () {
Object fps;
fps = Instantiate(EditorUtility.FindAsset(
"Standard Assets/ Prefabs/First Person Controller.prefab",
typeof(Transform)));
Selection.activeObject = fps;
}
